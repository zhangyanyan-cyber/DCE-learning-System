离散选择实验 (DCE)

一种基于随机效用理论的偏好测量方法，通过构建假想情景让受访者在不同选项间进行权衡，从而量化属性相对重要性与支付意愿。

核心概念

DCE 源于 Lancaster (1966) 消费者理论与 McFadden (1974) 随机效用理论，目前广泛应用于医疗、环境、交通等多个领域。

🎯

属性 (Attribute)

描述选项特征的维度，例如"治疗费用"、"副作用发生率"等。每个 DCE 通常含 4–8 个属性。

📏

水平 (Level)

属性的具体取值，例如费用的水平为"100元、300元、500元"。每个属性通常设 2–5 个水平。

📋

选择集 (Choice Set)

受访者在一次选择任务中面对的选项组合，通常由 2–3 个备选方案构成，每人回答 8–16 个选择集。

🔧

实验设计 (Design)

决定属性水平如何组合成选择集的方案，常见方法包括正交设计、最大化设计和 D-optimal 设计。

💰

支付意愿 (WTP)

受访者愿意为某属性水平变化支付的最大金额，通过效用参数估计计算得出。

📈

随机效用模型

U = V + ε，其中 V 为可观测效用，ε 为误差项。不同误差分布假设对应不同模型形式。

→

属性与水平的选择通常需结合文献综述、专家咨询、定性访谈（焦点小组）来完成，是整个研究的基础。

📌 效用函数

Vnij = β₁X₁ + β₂X₂ + ... + βₖXₖ
P(i|C) = exp(Vni) / Σⱼ exp(Vnj)

β 系数代表属性对效用的边际贡献，通过最大似然估计得到。

📌 WTP 计算

WTP = −(βattribute / βcost)
CI 通过 Krinsky-Robb 法或 Delta 法估计

价格属性系数符号应为负，其他属性的 WTP 单位与价格单位一致。

问卷设计 R 语言代码

以下提供三种实验设计方法的完整 R 代码，以医疗卫生场景（3属性×3水平）为示例，可根据实际情况修改属性和水平。

通用准备： 在运行以下任何设计代码前，请先安装所需的 R 包：
install.packages(c("AlgDesign", "DoE.base", "idefix", "support.CEs"))

01 正交设计 (Orthogonal Design)

正交设计确保属性之间相互正交（不相关），是 DCE 最传统、应用最广泛的设计方法。使用 DoE.base 包和 support.CEs 包实现。

# =============================================================
# DCE 正交实验设计
# 场景：医疗卫生 - 治疗方案偏好研究
# 属性：治疗效果(3)、副作用(3)、费用(4)
# =============================================================

library(DoE.base)
library(support.CEs)

# ---- Step 1: 定义属性与水平 ----
# 属性名称与水平数目
attr_levels <- c(
  treatment_effect = 3,   # 水平：50%, 70%, 90%
  side_effect      = 3,   # 水平：无, 轻度, 中度
  cost             = 4    # 水平：200, 500, 1000, 2000元
)

# 生成全因子（全部可能组合）
full_factorial <- expand.grid(
  treatment_effect = c("50%", "70%", "90%"),
  side_effect      = c("无", "轻度", "中度"),
  cost             = c("200元", "500元", "1000元", "2000元")
)
cat("全因子总数:", nrow(full_factorial), "\n")  # 36个组合

# ---- Step 2: 生成正交主效应计划 ----
# oa.design 生成正交阵
orth_design <- oa.design(
  nlevels  = attr_levels,
  columns  = "min3",    # 最小混杂列分配
  replications = 1,
  randomize    = FALSE
)

cat("\n正交设计方案（属性水平编码）:\n")
print(orth_design)
cat("正交设计行数:", nrow(orth_design), "\n")

# ---- Step 3: 替换为实际水平标签 ----
design_labeled <- orth_design
levels(design_labeled$treatment_effect) <- c("50%", "70%", "90%")
levels(design_labeled$side_effect)      <- c("无", "轻度", "中度")
levels(design_labeled$cost)             <- c("200元", "500元", "1000元", "2000元")

cat("\n实际标签正交设计:\n")
print(design_labeled)

# ---- Step 4: 构建选择集（使用 support.CEs 包）----
# 将正交设计分组为选择集，每个选择集含2个备选方案
n_alternatives <- 2   # 每个选择集中的选项数
n_questions    <- 9   # 每位受访者回答的问题数

# 生成选择集
choice_sets <- rotation.design(
  attribute.names = list(
    treatment_effect = c("50%", "70%", "90%"),
    side_effect      = c("无", "轻度", "中度"),
    cost             = c("200元", "500元", "1000元", "2000元")
  ),
  nalternatives = n_alternatives,
  nblocks       = 1,
  row.renames   = FALSE,
  randomize     = TRUE,
  seed          = 42
)

cat("\n选择集设计:\n")
print(choice_sets)

# ---- Step 5: 检验正交性 ----
# 检查属性间相关性（理想情况下应接近0）
corr_check <- cor(data.matrix(design_labeled))
cat("\n属性间相关矩阵（正交设计应接近单位矩阵）:\n")
print(round(corr_check, 3))

# ---- Step 6: 导出设计用于问卷制作 ----
write.csv(design_labeled, "orthogonal_design.csv",
          row.names = FALSE, fileEncoding = "UTF-8")
cat("\n设计文件已保存至 orthogonal_design.csv\n")

02 最大化设计 (Maximal / Full Factorial Fractional Design)

最大化设计在样本量约束下最大化信息效率，通过 AlgDesign 包的 optFederov 函数实现，适合属性水平较多、全因子不可行的情况。

# =============================================================
# DCE 最大化设计（Maximal / Federov Exchange Algorithm）
# 适用于：寻找估计能力最优的分式因子设计
# =============================================================

library(AlgDesign)

# ---- Step 1: 构建候选集（全因子空间）----
treatment_effect <- c(50, 70, 90)      # 治愈率 (%)
side_effect      <- c(1, 2, 3)         # 1=无, 2=轻度, 3=中度
cost             <- c(200, 500, 1000, 2000)  # 费用（元/月）
waiting_time     <- c(1, 7, 30)        # 等待时间（天）

# 生成全因子候选集
candidate_set <- expand.grid(
  treatment_effect = treatment_effect,
  side_effect      = side_effect,
  cost             = cost,
  waiting_time     = waiting_time
)
cat("候选集大小:", nrow(candidate_set), "\n")

# ---- Step 2: 构建模型矩阵（主效应 + 可选交互项）----
# 主效应模型
model_formula <- ~ treatment_effect + side_effect + cost + waiting_time

# 若需考虑交互效应（谨慎使用，会增加所需行数）:
# model_formula <- ~ (treatment_effect + side_effect + cost + waiting_time)^2

model_matrix <- model.matrix(model_formula, data = candidate_set)
cat("模型参数数量:", ncol(model_matrix), "\n")  # 包含截距

# ---- Step 3: Federov 交换算法最大化 D-效率 ----
# nTrials: 设计行数（建议为参数数量的 3-5 倍）
n_params <- ncol(model_matrix) - 1  # 不含截距
n_trials <- n_params * 4            # 行数 = 参数数×4

set.seed(2024)
optimal_result <- optFederov(
  frml    = model_formula,
  data    = candidate_set,
  nTrials = n_trials,
  criterion = "D",       # D-最优准则（最大化行列式）
  nRepeats  = 20,        # 重复次数，增大可提高质量
  nullify   = 1          # 处理不可估参数
)

cat("\n最优设计（前6行）:\n")
print(head(optimal_result$design))
cat("\nD-效率:", round(optimal_result$Dea, 4), "\n")
cat("设计行数:", nrow(optimal_result$design), "\n")

# ---- Step 4: 替换编码为水平标签 ----
design_max <- optimal_result$design
design_max$treatment_effect <- paste0(design_max$treatment_effect, "%")
design_max$side_effect <- factor(design_max$side_effect,
                                  levels = c(1,2,3),
                                  labels = c("无","轻度","中度"))
design_max$cost <- paste0(design_max$cost, "元/月")
design_max$waiting_time <- paste0(design_max$waiting_time, "天")

cat("\n标签化设计:\n")
print(design_max)

# ---- Step 5: 将设计行配对为选择集 ----
n_choice_sets <- floor(nrow(design_max) / 2)
choice_set_id <- rep(1:n_choice_sets, each = 2)
alt_id        <- rep(c("方案A", "方案B"), n_choice_sets)

# 若行数为奇数则截断
design_pairs <- design_max[1:(n_choice_sets*2), ]
design_pairs$choice_set <- choice_set_id
design_pairs$alternative <- alt_id

cat("\n配对选择集（前8行）:\n")
print(head(design_pairs, 8))

# ---- Step 6: 检验设计平衡性 ----
cat("\n各属性水平出现频次:\n")
for(col in c("treatment_effect","side_effect","cost","waiting_time")) {
  cat(col, ":\n")
  print(table(design_max[[col]]))
}

write.csv(design_pairs, "maximal_design.csv",
          row.names = FALSE, fileEncoding = "UTF-8")

03 D-optimal 设计

专为 DCE 选择情境优化的 D-optimal 设计，使用 idefix 包实现，直接针对多项 Logit 模型的信息矩阵进行优化，是目前方法论文献最推荐的设计方式。

# =============================================================
# DCE D-optimal 设计（针对多项Logit模型优化）
# 使用 idefix 包，直接最大化选择模型信息矩阵行列式
# =============================================================

library(idefix)

# ---- Step 1: 定义属性与水平 ----
# lvls: 每个属性的水平数
# coding: C = 对比编码, E = 效应编码, D = 虚拟编码
attrs <- list(
  treatment_effect = c("50%", "70%", "90%"),
  side_effect      = c("无", "轻度", "中度"),
  cost             = c("200元", "500元", "1000元", "2000元"),
  waiting_time     = c("1天", "1周", "1个月")
)

lvls   <- sapply(attrs, length)    # 各属性水平数: c(3,3,4,3)
coding <- rep("E", length(lvls))   # 效应编码(Effect coding)适合DCE

# 参数数量（效应编码下每个属性的参数数 = 水平数-1）
n_params <- sum(lvls - 1)
cat("模型参数数量（不含截距）:", n_params, "\n")

# ---- Step 2: 设定先验参数（零向量 = 设计局部最优；
#              若有先验估计，代入以提升效率）----
# 零先验（最保守，不依赖任何先验信息）
prior_mean <- rep(0, n_params)

# 若有文献先验（例如从小规模预实验或文献meta获取）:
# prior_mean <- c(0.8, 1.2, -0.5, -0.3, -1.0, -0.8, -1.5, 0.6, 0.4)

# ---- Step 3: 生成候选集 ----
cand_set <- Profiles(
  lvls   = lvls,
  coding = coding
)
cat("候选集大小（编码后）:", nrow(cand_set), "\n")

# ---- Step 4: D-optimal 协调算法 ----
# n_sets:    选择集数量
# n_alts:    每个选择集中的备选方案数
# start.des: 初始设计（NULL表示随机初始化）
n_sets <- 12   # 建议每位受访者回答10-16个选择集
n_alts <- 2    # 每个选择集含2个备选方案

set.seed(2024)
dopt_result <- CEA(
  lvls      = lvls,
  coding    = coding,
  par.draws = prior_mean,   # 使用零先验
  n.sets    = n_sets,
  n.alts    = n_alts,
  cand.set  = cand_set,
  n.start   = 12,           # 增加随机起点以逃离局部最优
  best      = TRUE          # 仅返回最优设计
)

cat("\nD-error（越小越好）:", round(dopt_result$error, 6), "\n")

# ---- Step 5: 解码与标签化 ----
# 将效应编码转换回水平标签
design_decoded <- Decode(
  des    = dopt_result$design,
  lvls   = lvls,
  coding = coding,
  alt.levs = attrs
)

cat("\n解码后选择集设计:\n")
print(design_decoded$design)

# ---- Step 6: 格式化为问卷格式 ----
design_survey <- design_decoded$design
design_survey$choice_set  <- rep(1:n_sets, each = n_alts)
design_survey$alternative <- rep(paste0("方案", LETTERS[1:n_alts]), n_sets)
design_survey$response    <- NA  # 预留受访者回答列

# 添加属性列名
colnames(design_survey)[1:length(attrs)] <- names(attrs)

cat("\n问卷格式（前6行）:\n")
print(head(design_survey))

# ---- Step 7: 贝叶斯 D-optimal（使用先验分布，更稳健）----
# 若要使用贝叶斯方法（BCEA函数），需设定先验分布
cat("\n---- 贝叶斯 D-optimal 设计 ----\n")
n_draws <- 100   # 蒙特卡洛抽样数（正式研究建议500+）

# 假设先验服从正态分布
prior_cov <- diag(0.5, n_params)  # 先验方差矩阵

set.seed(2024)
bayes_draws <- MASS::mvrnorm(
  n  = n_draws,
  mu = prior_mean,
  Sigma = prior_cov
)

bayes_result <- BCEA(
  lvls      = lvls,
  coding    = coding,
  par.draws = bayes_draws,
  n.sets    = n_sets,
  n.alts    = n_alts,
  cand.set  = cand_set,
  n.start   = 6
)

cat("贝叶斯 D-error:", round(bayes_result$error, 6), "\n")

# ---- Step 8: 比较两种 D-optimal 设计效率 ----
cat("\n设计效率比较:\n")
cat("局部 D-optimal D-error:", round(dopt_result$error, 6), "\n")
cat("贝叶斯 D-optimal D-error:", round(bayes_result$error, 6), "\n")

# 导出
write.csv(design_survey, "doptimal_design.csv",
          row.names = FALSE, fileEncoding = "UTF-8")
cat("\nD-optimal 设计已保存至 doptimal_design.csv\n")
三种设计方法对比： 正交设计简单易用但不针对选择模型优化；最大化设计（Federov）在全因子空间搜索信息最优组合；D-optimal 设计直接针对多项 Logit 模型的 Fisher 信息矩阵最大化，是方法论文献最推荐的方式。贝叶斯 D-optimal 进一步纳入先验不确定性，当研究者有先验估计时尤为适用。

数据分析 R 语言代码

包含数据准备、三种主流选择模型的估计代码，以及支付意愿（WTP）的计算与可视化方法。

数据准备与格式转换

DCE 数据需整理为"长格式（long format）"，每个备选方案占一行，并添加选择标识变量。

# =============================================================
# DCE 数据清洗与格式化
# 输入：宽格式调查数据（每行为一个选择任务）
# 输出：mlogit 包所需的长格式数据
# =============================================================

library(tidyverse)
library(mlogit)

# ---- Step 1: 读取原始数据 ----
# 假设原始数据格式：每行为一个受访者的一个选择任务
# 变量命名规则: attr_alt (例如 cost_A, cost_B, choice)

# 模拟数据示例（实际研究替换为 read.csv("your_data.csv")）
set.seed(42)
n_resp  <- 200  # 受访者数
n_tasks <- 12   # 每人回答12个选择集

raw_data <- expand.grid(
  respondent_id = 1:n_resp,
  task_id       = 1:n_tasks
) %>%
  mutate(
    # 方案A的属性水平（编码值）
    cost_A   = sample(c(200, 500, 1000, 2000), n(), replace = TRUE),
    effect_A = sample(c(50, 70, 90), n(), replace = TRUE),
    side_A   = sample(c(1, 2, 3), n(), replace = TRUE),
    # 方案B的属性水平
    cost_B   = sample(c(200, 500, 1000, 2000), n(), replace = TRUE),
    effect_B = sample(c(50, 70, 90), n(), replace = TRUE),
    side_B   = sample(c(1, 2, 3), n(), replace = TRUE),
    # 受访者选择（1=选A, 2=选B，基于效用模拟）
    choice   = sample(c(1, 2), n(), replace = TRUE, prob = c(0.55, 0.45))
  )

cat("原始数据（宽格式）前5行:\n")
print(head(raw_data, 5))

# ---- Step 2: 宽格式转长格式 ----
long_data <- raw_data %>%
  pivot_longer(
    cols      = matches("_(A|B)$"),
    names_to  = c(".value", "alternative"),
    names_sep = "_(?=[AB]$)"
  ) %>%
  mutate(
    # 创建选择标识变量（1=被选中）
    chosen = as.integer(
      (alternative == "A" & choice == 1) |
      (alternative == "B" & choice == 2)
    ),
    # 确保变量类型正确
    alternative   = factor(alternative),
    respondent_id = factor(respondent_id),
    # 属性编码为因子（用于效应编码）或保留数值
    cost   = as.numeric(cost),
    effect = as.numeric(effect),
    side   = factor(side, levels = 1:3, labels = c("无","轻度","中度"))
  ) %>%
  arrange(respondent_id, task_id, alternative)

cat("\n长格式数据前10行:\n")
print(head(long_data, 10))
cat("\n数据维度:", nrow(long_data), "行 ×", ncol(long_data), "列\n")

# ---- Step 3: 数据完整性检查 ----
cat("\n-- 数据质量检查 --\n")
# 每个选择集应恰好有1行被选中
check_choice <- long_data %>%
  group_by(respondent_id, task_id) %>%
  summarise(n_chosen = sum(chosen), .groups = "drop")

cat("选择任务中被选中数量分布（应全为1）:\n")
print(table(check_choice$n_chosen))

# 检查缺失值
cat("\n各列缺失值数量:\n")
print(colSums(is.na(long_data)))

# ---- Step 4: 转为 mlogit.data 格式 ----
mlogit_data <- mlogit.data(
  long_data,
  choice     = "chosen",
  shape      = "long",
  alt.var    = "alternative",
  id.var     = "respondent_id",
  chid.var   = "task_id"
)

cat("\nmlogit 格式数据结构:\n")
str(mlogit_data)

# ---- Step 5: 可选 - 添加受访者特征 ----
# 若有受访者社会经济特征，合并入主数据
respondent_chars <- data.frame(
  respondent_id = factor(1:n_resp),
  age     = sample(18:70, n_resp, replace = TRUE),
  gender  = sample(c("男","女"), n_resp, replace = TRUE),
  income  = sample(c("低","中","高"), n_resp, replace = TRUE),
  edu     = sample(c("高中及以下","大专/本科","研究生"), n_resp, replace = TRUE)
)

long_data_full <- long_data %>%
  left_join(respondent_chars, by = "respondent_id")

cat("\n含受访者特征的数据（前3行）:\n")
print(head(long_data_full, 3))

# 保存清洁数据
save(long_data, mlogit_data, respondent_chars, file = "dce_data.RData")
cat("\n数据已保存至 dce_data.RData\n")

条件 Logit 模型 (CL / MNL)

最基础的 DCE 模型，假设参数在所有受访者间相同（固定效用），使用 mlogit 包估计。

# =============================================================
# 条件 Logit / 多项 Logit 模型 (CL/MNL)
# 假设：IIA (无关选项独立性)，固定参数，同质偏好
# =============================================================

library(mlogit)
library(tidyverse)

load("dce_data.RData")  # 加载准备好的数据

# ---- Step 1: 基本 CL 模型（主效应）----
cl_model <- mlogit(
  chosen ~ cost + effect + side | 0,  # |0 表示不含备选方案特定常数 (ASC)
  data  = mlogit_data,
  shape = "long",
  alt.var = "alternative"
)

cat("=== 条件 Logit 模型结果 ===\n")
summary(cl_model)

# 提取关键指标
cat("\n模型拟合指标:\n")
loglik_null  <- -nrow(mlogit_data)/2 * log(2)  # 等概率模型对数似然
loglik_model <- as.numeric(cl_model$logLik)
mcfadden_r2  <- 1 - loglik_model / loglik_null
cat("McFadden R²:", round(mcfadden_r2, 4), "\n")
cat("AIC:", AIC(cl_model), "\n")
cat("BIC:", BIC(cl_model), "\n")

# ---- Step 2: 含 ASC 的模型（适用于包含"不选"选项的设计）----
# 若问卷含第三个选项（如"两个都不选"），取消注释
# cl_model_asc <- mlogit(
#   chosen ~ cost + effect + side | 1,  # |1 加入 ASC
#   data = mlogit_data, reflevel = "B"
# )

# ---- Step 3: 含交互项的模型 ----
cl_model_int <- mlogit(
  chosen ~ cost + effect + side + cost:effect | 0,
  data = mlogit_data
)

# 模型比较（似然比检验）
lrtest_result <- lrtest(cl_model, cl_model_int)
cat("\n主效应 vs 含交互项模型（LR检验）:\n")
print(lrtest_result)

# ---- Step 4: 提取系数与置信区间 ----
coef_table <- data.frame(
  parameter = names(coef(cl_model)),
  estimate  = coef(cl_model),
  std_error = sqrt(diag(vcov(cl_model))),
  z_value   = coef(cl_model) / sqrt(diag(vcov(cl_model)))
) %>%
  mutate(
    p_value  = 2 * pnorm(-abs(z_value)),
    ci_lower = estimate - 1.96 * std_error,
    ci_upper = estimate + 1.96 * std_error,
    sig      = case_when(
      p_value < 0.001 ~ "***",
      p_value < 0.01  ~ "**",
      p_value < 0.05  ~ "*",
      p_value < 0.1   ~ ".",
      TRUE            ~ ""
    )
  )

cat("\n系数估计与置信区间:\n")
print(coef_table %>% select(-std_error) %>% mutate(across(where(is.numeric), ~round(.,4))))

# ---- Step 5: IIA 检验（Hausman-McFadden）----
# 去掉一个选项重新估计，检验 IIA 假设是否成立
cat("\n注：IIA 检验需在含3个以上选项的设计中使用 hmftest() 函数\n")

# ---- Step 6: 可视化系数 ----
library(ggplot2)
coef_plot_data <- coef_table %>%
  mutate(parameter = factor(parameter, levels = rev(parameter)))

p_coef <- ggplot(coef_plot_data, aes(x = estimate, y = parameter)) +
  geom_vline(xintercept = 0, linetype = "dashed", color = "gray50") +
  geom_errorbarh(aes(xmin = ci_lower, xmax = ci_upper),
                 height = 0.2, color = "#1a6b5a", linewidth = 0.8) +
  geom_point(size = 3, color = "#1a6b5a") +
  theme_minimal(base_size = 13) +
  labs(title = "条件 Logit 模型系数（95% CI）",
       x = "效用系数估计", y = "") +
  theme(panel.grid.minor = element_blank())

print(p_coef)
ggsave("cl_coefficients.png", p_coef, width = 8, height = 5, dpi = 150)

混合 Logit 模型 (MXL / Mixed Logit)

允许参数在受访者间随机变化（异质偏好），放松 IIA 假设，是目前 DCE 最常用的估计方法，使用模拟最大似然法（MSLE）估计。

# =============================================================
# 混合 Logit 模型 (Mixed Logit / Random Parameters Logit)
# 允许偏好异质性，通过随机系数捕捉个体差异
# =============================================================

library(mlogit)
library(mixl)    # 可选，提供更灵活的混合Logit估计
library(tidyverse)

load("dce_data.RData")

# ---- Step 1: 基础混合 Logit（正态分布随机参数）----
cat("估计混合 Logit 模型（可能需要2-5分钟）...\n")

mxl_normal <- mlogit(
  chosen ~ cost + effect + side | 0,
  data   = mlogit_data,
  panel  = TRUE,       # 使用面板数据结构（同一受访者多个任务）
  rpar   = c(          # 指定随机参数及其分布
    cost   = "n",      # 正态分布（注：价格参数有时用对数正态 "ln"）
    effect = "n",      # 正态分布
    side2  = "n",      # 轻度副作用（相对于基准"无"）
    side3  = "n"       # 中度副作用
  ),
  R      = 500,        # 抽样次数（越大越精确，建议正式分析用1000+）
  seed   = 42,
  halton = NA,         # 使用 Halton 序列（比随机抽样更高效）
  correlation = FALSE  # 是否估计随机参数间的相关性
)

cat("\n=== 混合 Logit 模型（独立随机参数）===\n")
summary(mxl_normal)

# ---- Step 2: 含相关随机参数的 MXL ----
cat("\n估计含相关结构的混合 Logit 模型...\n")
mxl_corr <- mlogit(
  chosen ~ cost + effect + side | 0,
  data        = mlogit_data,
  panel       = TRUE,
  rpar        = c(cost = "n", effect = "n", side2 = "n", side3 = "n"),
  R           = 500,
  seed        = 42,
  halton      = NA,
  correlation = TRUE   # 估计随机参数间协方差矩阵
)

summary(mxl_corr)

# Cholesky 分解矩阵（对角线 = 标准差，非对角线 = 相关系数）
cat("\n随机参数相关矩阵:\n")
print(round(cov2cor(cov.mlogit(mxl_corr)), 3))

# ---- Step 3: 对数正态分布（适用于价格参数，确保符号一致）----
mxl_lognorm <- mlogit(
  chosen ~ cost + effect + side | 0,
  data  = mlogit_data,
  panel = TRUE,
  rpar  = c(
    cost   = "ln",   # 对数正态（估计值为负，需加负号）
    effect = "n",
    side2  = "n",
    side3  = "n"
  ),
  R    = 500,
  seed = 42
)
summary(mxl_lognorm)

# ---- Step 4: 提取均值和标准差参数 ----
coef_mxl <- coef(mxl_normal)
# 均值参数（前半部分）和标准差参数（后半部分，前缀为"sd."）
mean_params <- coef_mxl[!grepl("^sd\\.", names(coef_mxl))]
sd_params   <- coef_mxl[grepl("^sd\\.", names(coef_mxl))]

cat("\n随机参数均值:\n"); print(round(mean_params, 4))
cat("\n随机参数标准差:\n"); print(round(abs(sd_params), 4))

# ---- Step 5: 条件（个体）参数估计 ----
# 利用贝叶斯定理估计每位受访者的条件期望参数
# 需要受访者标识符与模型兼容
individual_params <- fitted(mxl_normal, type = "parameters")
cat("\n个体条件参数估计（前6行）:\n")
print(head(individual_params, 6))

# 可视化个体参数分布
df_ind <- as.data.frame(individual_params)
p_het <- ggplot(df_ind, aes(x = cost)) +
  geom_histogram(bins = 30, fill = "#1a6b5a", alpha = 0.7, color = "white") +
  geom_vline(xintercept = mean(df_ind$cost), linetype = "dashed", color = "#c4832a") +
  theme_minimal(base_size = 13) +
  labs(title = "费用属性个体条件参数分布",
       x = "费用参数估计值", y = "频次") +
  theme(panel.grid.minor = element_blank())
print(p_het)

# ---- Step 6: 模型比较 ----
cat("\n=== 模型比较 ===\n")
models_aic <- data.frame(
  Model = c("条件Logit", "MXL(独立)", "MXL(相关)"),
  AIC   = c(AIC(cl_model), AIC(mxl_normal), AIC(mxl_corr)),
  BIC   = c(BIC(cl_model), BIC(mxl_normal), BIC(mxl_corr)),
  LogLik = c(logLik(cl_model), logLik(mxl_normal), logLik(mxl_corr))
)
print(models_aic %>% mutate(across(where(is.numeric), ~round(.,2))))

潜在类别模型 (LCM)

将样本划分为偏好结构不同的若干潜在类别，同一类别内参数固定，类别成员资格由概率决定。使用 gmnl 包实现。

# =============================================================
# 潜在类别模型 (Latent Class Model, LCM)
# 捕捉受访者间的离散偏好异质性
# =============================================================

library(gmnl)
library(tidyverse)

load("dce_data.RData")

# ---- Step 1: 确定最优类别数（BIC准则）----
cat("正在估计不同类别数的 LCM 模型...\n")

bic_results <- data.frame(
  classes = integer(),
  loglik  = numeric(),
  n_params = integer(),
  AIC     = numeric(),
  BIC     = numeric()
)

for (k in 2:5) {
  cat("估计", k, "类模型...\n")
  tryCatch({
    lcm_k <- gmnl(
      chosen ~ cost + effect + side | 0 | 0 | 0 | 1,
      data   = mlogit_data,
      model  = "lc",
      Q      = k,        # 类别数
      panel  = TRUE,
      method = "bhhh",   # BHHH 优化算法
      iterlim = 500,
      print.level = 0
    )
    n_par <- length(coef(lcm_k))
    n_obs <- nrow(mlogit_data) / 2
    bic_results <- rbind(bic_results, data.frame(
      classes  = k,
      loglik   = as.numeric(logLik(lcm_k)),
      n_params = n_par,
      AIC      = -2 * logLik(lcm_k) + 2 * n_par,
      BIC      = -2 * logLik(lcm_k) + log(n_obs) * n_par
    ))
  }, error = function(e) cat("  类别数", k, "估计失败:", e$message, "\n"))
}

cat("\n=== 不同类别数的模型比较 ===\n")
print(bic_results %>% mutate(across(where(is.numeric), ~round(.,2))))
cat("建议选择 BIC 最小的类别数\n")

# ---- Step 2: 估计最优类别数模型（以3类为例）----
best_k <- bic_results$classes[which.min(bic_results$BIC)]
cat("\n根据 BIC，最优类别数为:", best_k, "\n")

set.seed(42)
lcm_best <- gmnl(
  chosen ~ cost + effect + side | 0 | 0 | 0 | 1,
  data   = mlogit_data,
  model  = "lc",
  Q      = best_k,
  panel  = TRUE,
  method = "bhhh",
  iterlim = 500
)

cat("\n=== 潜在类别模型结果 ===\n")
summary(lcm_best)

# ---- Step 3: 提取各类别参数与类别规模 ----
coef_lcm <- coef(lcm_best)

# 分离类别参数和类别成员概率参数
class_coefs <- matrix(coef_lcm[!grepl("^class", names(coef_lcm))],
                       ncol = best_k)
colnames(class_coefs) <- paste0("Class ", 1:best_k)
rownames(class_coefs) <- c("cost", "effect", "side轻度", "side中度")

cat("\n各类别效用系数:\n")
print(round(class_coefs, 4))

# 类别规模（成员概率）
class_probs <- lcm_best$prob.class
cat("\n各类别成员比例:\n")
print(round(colMeans(class_probs), 3))

# ---- Step 4: 个体类别归属预测 ----
# 根据最大后验概率分配每位受访者到类别
class_assignment <- apply(class_probs, 1, which.max)
class_dist <- table(class_assignment)
cat("\n样本类别分配:\n")
print(class_dist)
cat("各类别占比:", round(prop.table(class_dist) * 100, 1), "%\n")

# ---- Step 5: 解释类别特征 ----
# 结合受访者特征分析各类别的社会经济画像
# 注：需将 respondent_chars 与类别分配合并

# 类别 WTP 计算
wtp_by_class <- data.frame(
  Class = paste0("Class ", 1:best_k)
)

for (k in 1:best_k) {
  wtp_by_class[k, "WTP_effect"] <- -class_coefs["effect", k] / class_coefs["cost", k]
  wtp_by_class[k, "WTP_side_light"] <- -class_coefs["side轻度", k] / class_coefs["cost", k]
  wtp_by_class[k, "WTP_side_mod"] <- -class_coefs["side中度", k] / class_coefs["cost", k]
}

cat("\n各类别支付意愿（元/月）:\n")
print(round(wtp_by_class, 2))

# ---- Step 6: 类别偏好雷达图可视化 ----
library(ggplot2)
class_coefs_df <- as.data.frame(t(class_coefs)) %>%
  rownames_to_column("class") %>%
  pivot_longer(-class, names_to = "attribute", values_to = "utility")

p_class <- ggplot(class_coefs_df, aes(x = attribute, y = utility,
                                       fill = class, group = class)) +
  geom_col(position = "dodge", alpha = 0.85) +
  geom_hline(yintercept = 0, linewidth = 0.5) +
  scale_fill_manual(values = c("#1a6b5a","#c4832a","#3b5fa0")) +
  theme_minimal(base_size = 13) +
  labs(title = "各潜在类别效用系数比较",
       x = "属性", y = "效用系数", fill = "类别") +
  theme(panel.grid.minor = element_blank(),
        legend.position = "top")
print(p_class)
ggsave("lcm_classes.png", p_class, width = 9, height = 5, dpi = 150)

支付意愿 (WTP) 计算与可视化

从模型系数计算 WTP 值，并使用 Krinsky-Robb 法（参数 Bootstrap）构建置信区间，最后进行森林图可视化。

# =============================================================
# 支付意愿 (WTP) 计算与不确定性分析
# 方法：Krinsky-Robb 参数 Bootstrap（推荐）
# =============================================================

library(mlogit)
library(MASS)
library(tidyverse)

load("dce_data.RData")

# ---- Step 1: 估计模型（以条件Logit为例）----
cl_model <- mlogit(
  chosen ~ cost + effect + side | 0,
  data = mlogit_data
)

# ---- Step 2: 点估计 WTP ----
b <- coef(cl_model)
b_cost <- b["cost"]  # 费用系数（应为负值）

# WTP = -(非价格属性系数) / (价格属性系数)
wtp_point <- data.frame(
  attribute   = c("效果提升 (+1%)", "副作用:轻度 vs 无", "副作用:中度 vs 无"),
  wtp_estimate = c(
    -b["effect"]   / b_cost,
    -b["side轻度"] / b_cost,
    -b["side中度"] / b_cost
  )
)
cat("=== WTP 点估计（元/月）===\n")
print(round(wtp_point, 2))

# ---- Step 3: Krinsky-Robb 置信区间 ----
n_draws <- 10000   # Bootstrap 抽样次数（建议10000以上）
set.seed(2024)

# 从参数的渐近正态分布中抽样
coef_draws <- mvrnorm(
  n     = n_draws,
  mu    = coef(cl_model),
  Sigma = vcov(cl_model)
)
head(coef_draws)

# 计算每次抽样的 WTP
wtp_draws <- data.frame(
  WTP_effect     = -coef_draws[,"effect"]   / coef_draws[,"cost"],
  WTP_side_light = -coef_draws[,"side轻度"] / coef_draws[,"cost"],
  WTP_side_mod   = -coef_draws[,"side中度"] / coef_draws[,"cost"]
)

# 去除极端值（可选：Poe et al. 截断法）
wtp_clean <- wtp_draws %>%
  filter(
    between(WTP_effect,     quantile(WTP_effect,     0.01), quantile(WTP_effect,     0.99)),
    between(WTP_side_light, quantile(WTP_side_light, 0.01), quantile(WTP_side_light, 0.99)),
    between(WTP_side_mod,   quantile(WTP_side_mod,   0.01), quantile(WTP_side_mod,   0.99))
  )

# 计算均值和 95% CI
wtp_ci <- wtp_clean %>%
  summarise(across(everything(), list(
    mean  = mean,
    lower = ~quantile(., 0.025),
    upper = ~quantile(., 0.975)
  ))) %>%
  pivot_longer(everything(),
               names_to  = c("attribute", ".value"),
               names_sep = "_(?=mean|lower|upper)")

cat("\n=== WTP 估计（95% CI）===\n")
print(round(wtp_ci, 2))

# ---- Step 4: 完整 WTP 汇总表 ----
wtp_table <- wtp_ci %>%
  mutate(
    attribute = case_when(
      attribute == "WTP_effect"     ~ "治疗效果提升1%",
      attribute == "WTP_side_light" ~ "轻度副作用（vs无）",
      attribute == "WTP_side_mod"   ~ "中度副作用（vs无）"
    ),
    label = sprintf("%.1f元 (%.1f, %.1f)", mean, lower, upper)
  )
cat("\n=== WTP 汇总（元/月，95% CI）===\n")
print(wtp_table %>% select(attribute, label))

# ---- Step 5: WTP 森林图 ----
